\noindent $\bullet$  The {\sl conjugate transpose} or {\sl Hermitian transpose} of an $m \times n$ complex matrix $A$ is the $n \times m$ matrix $A^H = \overline{A}^T$ obtained from $A$ by taking the complex conjugate of each entry and then taking the transpose. 
\vskip.05in
\noindent $\bullet$ A complex matrix $S$ is a  \emph{Hermitian matrix} if $S=S^H$.

\vskip.1in

\noindent{\bf 2. }  Use properties of the transpose of a matrix and complex arithmetic to prove that  $A^HA$ is always a Hermitian matrix.

\begin{proof}
    Let $A = [z_{ij}]$ be an $m \times n$ matrix.
    Then, the $ij^{\text{th}}$ entry in $A^HA$, denoted $a_{ij}$, is $$\sum\limits_{k=1}^n \overline{z_{ik}}z_{jk}.$$
    Further, the $ji^{\text{th}}$ entry in $A^HA$, denoted $a_{ji}$, is
    \begin{align*}
        \sum\limits_{k=1}^n \overline{z_{jk}}z_{ik} &= \sum\limits_{k=1}^n \overline{z_{jk}\overline{z_{ik}}} \\
        &= \overline{\sum\limits_{k=1}^n z_{jk}\overline{z_{ik}}} \\
        &= \overline{\sum\limits_{k=1}^n \overline{z_{ik}}z_{jk}} \\
        &= \overline{a_{ij}}.
    \end{align*}
    Thus, $A^HA = (A^HA)^H$, so $A^HA$ is a Hermitian matrix.

\end{proof}
